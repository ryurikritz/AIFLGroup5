2. Feature Engineering

Deriving 10 new features from raw ticket text and metadata.

# --- TEXT FEATURES --- df['body_length'] = df['Body'].str.len() df['word_count'] = df['Body'].str.split().str.len() df['sentence_count'] = df['Body'].str.count(r'[.!?]+') df['avg_word_length'] = df['Body'].apply(lambda x: np.mean([len(w) for w in x.split()]) if x else 0) df['has_question'] = df['Body'].str.contains(r'\?').astype(int) # --- URGENCY / SENTIMENT FEATURES --- df['urgency_keywords'] = df['Body'].str.lower().str.count(r'urgent|immediately|critical|outage|down|broken|asap') df['politeness_score'] = df['Body'].str.lower().str.count(r'please|thank|kindly|appreciate') # --- LABEL ENCODING --- df['priority_numeric'] = df['Priority'].map({'low':1, 'medium':2, 'high':3}) # --- TAG FEATURES --- def parse_tags(t): try: return ast.literal_eval(t) except: return [] df['tags_list'] = df['Tags'].apply(parse_tags) df['tag_count'] = df['tags_list'].apply(len) df['has_IT_tag'] = df['tags_list'].apply(lambda t: int('IT' in t)) df['has_outage_tag']= df['tags_list'].apply(lambda t: int('Outage' in t)) print("Feature engineering complete!") print(f"Total features: {len(df.columns)} (5 original + 11 engineered)") feature_summary = { 'body_length':'Character count of ticket body', 'word_count':'Token count of ticket body', 'sentence_count':'Number of sentences (punctuation-based)', 'avg_word_length':'Mean word length — proxy for technical complexity', 'has_question':'Binary: ticket contains a question mark', 'urgency_keywords':'Count of urgency-signalling words', 'politeness_score':'Count of polite/formal markers', 'priority_numeric':'Ordinal encoding: low=1, medium=2, high=3', 'tag_count':'Number of tags attached to ticket', 'has_IT_tag':'Binary: IT tag present', 'has_outage_tag':'Binary: Outage tag present' } for k,v in feature_summary.items(): print(f" {k:22s}: {v}")

4. LangGraph Confidence Routing Simulation

# Simulate how Cortex ML confidence scores route tickets # Based on observed feature patterns in the dataset np.random.seed(42) n = len(df) # Confidence proxy: urgency + priority signal conf_base = 0.60 + (df['priority_numeric'] / 3) * 0.25 + df['urgency_keywords'].clip(0,3) * 0.04 conf_noise = np.random.normal(0, 0.06, n) conf_scores = (conf_base + conf_noise).clip(0, 1) df['confidence_score'] = conf_scores autonomous = (conf_scores >= 0.90).sum() hitl = ((conf_scores >= 0.70) & (conf_scores < 0.90)).sum() escalate = (conf_scores < 0.70).sum() print("=== LangGraph Routing Simulation ===") print(f"Total tickets : {n:,}") print(f"Autonomous (≥0.90) : {autonomous:,} ({autonomous/n*100:.1f}%)") print(f"HITL (0.70–0.89) : {hitl:,} ({hitl/n*100:.1f}%)") print(f"Escalate (<0.70) : {escalate:,} ({escalate/n*100:.1f}%)")
